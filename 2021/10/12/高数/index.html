常见不定积分

$$
\int x^kdx=\frac{1}{k+1}x^{k+1}+C,k\neq-1
$$

$$
\int \frac 1 xdx=ln|x|+C
$$

$$
\int e^xdx=e^x+C;\int a^xdx=\frac {a^x} {lna}+C,a>0 \ and\ a \neq 1
$$

$$
\int tanxdx=-ln|cosx|+C;\int cot xdx = ln |sinx|+C
$$

$$
\int \frac {dx} {cosx}=\int secxdx=ln|secx+tanx|+C
$$

$$
\int \frac {dx} {sinx}=\int cscxdx=ln|cscx-cotx|+C
$$

$$
\int sec^2xdx=tanx+C;\int csc^2x=-cot+C
$$

$$
\int secx\ tanxdx=secx+C;\int csc x\ cotxdx=-cscx+C
$$

$$
\int \frac {1} {a^2+x^2}dx = \frac 1 a arctan\frac x a+C
$$

$$
\int \frac 1 {\sqrt {a^2-x^2}}=arcsin\frac x a+C
$$

$$
\begin{cases}
\int \frac 1 {\sqrt {x^2+a^2}}=ln(x+\sqrt {x^2+a^2})+C \
\int \frac 1 {\sqrt {x^2-a^2}}=ln|x+\sqrt {x^2-a^2}|+C \
\end{cases}
$$

$$
\int \frac 1 {x^2-a^2}dx=\frac 1 {2a}ln|\frac {x-a}{x+a}|+C
$$

$$
\int \sqrt {a^2-x^2}dx=\frac {a^2} 2 arcsin\frac x a+\frac x 2 \sqrt {a^2-x^2}+C
$$

$$
\int sin^2dx=\frac x 2 - \frac {sin2x} 4+C
$$

$$
\int cos^2dx=\frac x 2 + \frac {sin2x} 4+C
$$

$$
\int tan^2 xdx = tanx-x+C
$$

$$
\int cot^2 xdx = -cotx-x+C
$$
